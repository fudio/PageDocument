 protected void btn_login_Click(object sender, EventArgs e)
        {
            Session["username"] = exampleInputName.Text;
            Session["pass"] = exampleInputPassword.Text;
            string ten = Session["username"].ToString();
            string pass = Session["pass"].ToString();
            string passmd5 = StringProc.MD5Hash(pass);
            if (Member.ktDangNhap(ten,passmd5))
            {
                Response.Redirect("refer.aspx");
            }
            else
            {
                Response.Write("Đăng Nhập Thất Bại");
            }
        }

            if(DataProvider.KTTK(query))

        }

        {
